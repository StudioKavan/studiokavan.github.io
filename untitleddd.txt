<script>
  const items = document.querySelectorAll('.item');
  const totalEl = document.getElementById('total');

  items.forEach(item => {
    item.addEventListener('change', () => {
      let total = 0;
      items.forEach(i => {
        if(i.checked) total += parseFloat(i.dataset.price);
      });
      totalEl.textContent = total;
    });
  });
</script>
